add_executable(prog
    main.cpp
    App.cpp
    Window.cpp
    Input.cpp
    Camera.cpp
    MapRenderer.cpp
    Data.cpp
    ResourceManager.cpp
    Texture.cpp
    glad.c
)

# Resolve IXWebSocket target name before linking
if(TARGET ixwebsocket::ixwebsocket)
    set(IXWS_TARGET ixwebsocket::ixwebsocket)
elseif(TARGET ixwebsocket)
    set(IXWS_TARGET ixwebsocket)
else()
    message(FATAL_ERROR "IXWebSocket target not found")
endif()

# Resolve nlohmann_json target name before linking
if(TARGET nlohmann_json::nlohmann_json)
    set(JSON_TARGET nlohmann_json::nlohmann_json)
elseif(TARGET nlohmann_json)
    set(JSON_TARGET nlohmann_json)
else()
    message(FATAL_ERROR "nlohmann_json target not found")
endif()

target_link_libraries(prog PRIVATE
    project_headers
    glm::glm
    ${JSON_TARGET}
    ${IXWS_TARGET}
    ${CMAKE_DL_LIBS}
)

if(TARGET SDL2::SDL2)
    target_link_libraries(prog PRIVATE SDL2::SDL2)
elseif(TARGET SDL2::SDL2-static)
    target_link_libraries(prog PRIVATE SDL2::SDL2-static)
else()
    target_link_libraries(prog PRIVATE SDL2)
endif()

# Keep runtime assets close to the executable for easy local runs.
add_custom_command(TARGET prog POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        ${PROJECT_SOURCE_DIR}/shaders
        $<TARGET_FILE_DIR:prog>/shaders
)
